Database Schema:
  uploaded_files:
    - id (UUID primary key)
    - created_at (timestamp)
    - file_name, file_path, file_size, file_type
    - storage_bucket (ra-presentation | correspondence | supporting-evidence)
    - uploaded_by (tracks who uploaded)
    - project_name (E-Government Payment Gateway)
    - ra_id (RA-2026-0042)
    - metadata (JSON: public_url, etc.)

  risk_acceptances:
    - ra_id (RA-2026-0042)
    - project_name
    - vulnerabilities (JSON array)
    - proposed_controls (text array)
    - assessment_notes (text array)
    - recommendation (APPROVE/REJECT)
    - status (pending/under_review/approved/rejected)

  projects:
    - project_name (E-Government Payment Gateway)
    - description
    - system_type (3-tier-web-application)
    - architecture_diagram_url
